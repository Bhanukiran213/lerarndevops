---
- hosts: all
  tasks:
  - name: Installing Apache using playbook
    yum:
      name: httpd
      state: installed
    become: yes
  - name: starting Apache
    service:
      name: httpd
      state: started
    become: yes
  - name: copying to /var/www/html
    copy:
      src: "/home/ec2-user/index.html"
      dest: "/var/www/html"
    become: yes
